# Fractal Tree
clear screen
clear graphics

# Rectangular leaf anchored at the bottom "stem"
define procedure drawLeaf(center, baseAngle) (
    set color list(random(30, 90), random(80, 100), random(10, 50))
    
    put random(5, 9) in leafWidth
    put random(6, 12) in leafHeight
    
    # Total rotation in radians
    put (baseAngle + random(-20, 20)) * pi / 180 in angleRad
    put cos(angleRad) in c
    put sin(angleRad) in s

    # Local rectangle anchored at the bottom
    put -leafWidth / 2 in x1_loc
    put 0 in y1_loc
    put leafWidth / 2 in x2_loc
    put leafHeight in y2_loc

    # Four vertices after rotation + translation
    put (x1_loc * c - y1_loc * s + x of center) in v1x
    put (x1_loc * s + y1_loc * c + y of center) in v1y
    put (x2_loc * c - y1_loc * s + x of center) in v2x
    put (x2_loc * s + y1_loc * c + y of center) in v2y
    put (x2_loc * c - y2_loc * s + x of center) in v3x
    put (x2_loc * s + y2_loc * c + y of center) in v3y
    put (x1_loc * c - y2_loc * s + x of center) in v4x
    put (x1_loc * s + y2_loc * c + y of center) in v4y

    triangle with vertices point(v1x, v1y), point(v2x, v2y), point(v3x, v3y)
    triangle with vertices point(v1x, v1y), point(v4x, v4y), point(v3x, v3y)
    
    # Short stem aligned with the leaf
    set color list(50, 30, 15)
    put (0 * c - (-2) * s + x of center) in s2x
    put (0 * s + (-2) * c + y of center) in s2y
    line thickness 1 from center to point(s2x, s2y)
)

# Recursive branch; leaves grow at the tip
define procedure drawBranch(startPos, branchLength, angle, level) (
    if (level > 0) then (
        put (angle * pi / 180) in angleRad
        put (x of startPos + branchLength * cos(angleRad)) in endX
        put (y of startPos + branchLength * sin(angleRad)) in endY
        put point(endX, endY) in endPos

        set color list(40, 26, 13)
        line thickness max(1, level / 2) from startPos to endPos
        
        put (0.75 + random(-10, 5) / 100) in lengthFactor
        put random(-15, 15) in var1
        put random(-15, 15) in var2
        
        drawBranch(endPos, branchLength * lengthFactor, angle - (25 + var1), level - 1)
        drawBranch(endPos, branchLength * lengthFactor, angle + (25 + var2), level - 1)
    )
    
    if (level == 0) then (
        # Leaf point moved forward along the branch and slightly to the side
        put (angle * pi / 180) in a
        put cos(a) in ca
        put sin(a) in sa
        put random(6, 12) in forward              # offset along the branch
        put random(-3, 3) in sideways               # lateral offset
        # perpendicular to the branch: (-sa, ca)
        put (x of startPos + forward * ca + sideways * (-sa)) in fx
        put (y of startPos + forward * sa + sideways * ca) in fy
        put point(fx, fy) in p_leaf
        
        # 2-4 leaves per tip, with small variations
        put random(2, 4) in n
        index i goes from 1 to n and you do (
            put angle + random(-15, 15) in a_loc
            # micro-shift to prevent overlapping
            put fx + random(-2, 2) in lx
            put fy + random(-2, 2) in ly
            drawLeaf(point(lx, ly), a_loc)
        )
    )
)

# Start
write "Drawing a fractal tree with natural leaves..."
put point(0, -220) in startPoint
drawBranch(startPoint, 90, 90, 9)

frame graphics
write "Done! Run again for variation."
