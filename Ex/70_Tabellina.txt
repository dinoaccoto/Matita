chiedi Numero dicendo "Inserisci un numero per la sua tabellina:"
cancella schermo
scrivi "Ecco la tabellina del " + Numero + ":"; accapo
scrivi "-------------------------"; accapo

variabile I va da 1 a 12 ed esegui (
    metti Numero * I in Risultato
    scrivi Numero + " x " + I + " = " + Risultato; accapo
)
