9. Fractale Boom

Een voorbeeld van recursie om een natuurlijke boom te tekenen.
Inclusief wiskundige functies (sinus, cosinus) om de vertakkingen te bereken.
Gebruikt willekeurige getallen om hoeken en lengtes te variëren, waardoor elke boom uniek is.

Matita 
# Fractale boom
wis scherm
wis canvas

# Rechthoekig blad verankerd aan de "stengel" onderaan
definieer procedure tekenBlad(centrum, basis_hoek) (
    stel kleur in (willekeurig(30, 90), willekeurig(80, 100), willekeurig(10, 50))
    
    zet willekeurig(5, 9) in breedte
    zet willekeurig(6, 12) in hoogte
    
    # Totale rotatie in radialen
    zet (basis_hoek + willekeurig(-20, 20)) * pi / 180 in hoek_rad
    zet cos(hoek_rad) in c
    zet sin(hoek_rad) in s

    # Lokale rechthoek verankerd onderaan
    zet -breedte / 2 in x1_lok
    zet 0 in y1_lok
    zet breedte / 2 in x2_lok
    zet hoogte in y2_lok

    # Vier hoekpunten na rotatie + translatie
    zet (x1_lok * c - y1_lok * s + x van centrum) in v1x
    zet (x1_lok * s + y1_lok * c + y van centrum) in v1y
    zet (x2_lok * c - y1_lok * s + x van centrum) in v2x
    zet (x2_lok * s + y1_lok * c + y van centrum) in v2y
    zet (x2_lok * c - y2_lok * s + x van centrum) in v3x
    zet (x2_lok * s + y2_lok * c + y van centrum) in v3y
    zet (x1_lok * c - y2_lok * s + x van centrum) in v4x
    zet (x1_lok * s + y2_lok * c + y van centrum) in v4y

    driehoek met hoekpunten punt(v1x, v1y), punt(v2x, v2y), punt(v3x, v3y)
    driehoek met hoekpunten punt(v1x, v1y), punt(v4x, v4y), punt(v3x, v3y)
    
    # Korte stengel uitgelijnd met het blad
    stel kleur in (50, 30, 15)
    zet (0 * c - (-2) * s + x van centrum) in s2x
    zet (0 * s + (-2) * c + y van centrum) in s2y
    lijn dikte 1 van centrum tot punt(s2x, s2y)
)

# Recursieve tak; bladeren groeien aan de punt
definieer procedure tekenTak(begin, takLengte, hoek, niveau) (
    als (niveau > 0) dan (
        zet (hoek * pi / 180) in hoek_rad
        zet (x van begin + takLengte * cos(hoek_rad)) in eind_x
        zet (y van begin + takLengte * sin(hoek_rad)) in eind_y
        zet punt(eind_x, eind_y) in eind_punt

        stel kleur in (40, 26, 13)
        lijn dikte max(1, niveau / 2) van begin tot eind_punt
        
        zet (0.75 + willekeurig(-10, 5) / 100) in lengte_factor
        zet willekeurig(-15, 15) in var1
        zet willekeurig(-15, 15) in var2
        
        tekenTak(eind_punt, takLengte * lengte_factor, hoek - (25 + var1), niveau - 1)
        tekenTak(eind_punt, takLengte * lengte_factor, hoek + (25 + var2), niveau - 1)
    )
    
    als (niveau == 0) dan (
        # Bladpunt naar voren verplaatst langs de tak en iets opzij
        zet (hoek * pi / 180) in a
        zet cos(a) in ca
        zet sin(a) in sa
        zet willekeurig(6, 12) in vooruit              # offset langs de tak
        zet willekeurig(-3, 3) in zijkant                # zijdelingse offset
        # loodrecht op de tak: (-sa, ca)
        zet (x van begin + vooruit * ca + zijkant * (-sa)) in fx
        zet (y van begin + vooruit * sa + zijkant * ca) in fy
        zet punt(fx, fy) in p_blad
        
        # 2-4 bladeren per top, met kleine variaties
        zet willekeurig(2, 4) in n
        teller i loopt van 1 tot n en je doet (
            zet hoek + willekeurig(-15, 15) in a_lok
            # micro-verschuiving om overlapping te voorkomen
            zet fx + willekeurig(-2, 2) in lx
            zet fy + willekeurig(-2, 2) in ly
            tekenBlad(punt(lx, ly), a_lok)
        )
    )
)

    globaal SPEELGEBIED, X_POS, Y_POS

definieer procedure update_spel () (
    globaal X_POS, Y_POS, AFSLUITEN, SNELHEID, LEVENS, GAME_OVER
    globaal Score, VOEDSEL_X, VOEDSEL_Y, VOEDSEL_GROOTTE, SPELER_STRAAL, DRIEHOEK_STRAAL, SPEELGEBIED
    globaal DRIEHOEK_MASSA, K_VEER, C_DEMPER, L_RUST, DT
    globaal LIJST_DRIEHOEK_X, LIJST_DRIEHOEK_Y, LIJST_DRIEHOEK_VX, LIJST_DRIEHOEK_VY

    als (toets "UP" is ingedrukt) dan ( zet Y_POS + SNELHEID in Y_POS )
    als (toets "DOWN" is ingedrukt) dan ( zet Y_POS - SNELHEID in Y_POS )
    als (toets "LEFT" is ingedrukt) dan ( zet X_POS - SNELHEID in X_POS )
    als (toets "RIGHT" is ingedrukt) dan ( zet X_POS + SNELHEID in X_POS )
    als (toets "X" is ingedrukt) dan ( zet 1 in AFSLUITEN )

    als (NUM_DRIEHOEKEN > 0) dan (
        teller i loopt van 1 tot NUM_DRIEHOEKEN en je doet (
            zet element i van LIJST_DRIEHOEK_X in T_X
            zet element i van LIJST_DRIEHOEK_Y in T_Y
            zet element i van LIJST_DRIEHOEK_VX in T_VX
            zet element i van LIJST_DRIEHOEK_VY in T_VY

            zet X_POS - T_X in VETTORE_X
            zet Y_POS - T_Y in VETTORE_Y
            zet wortel((VETTORE_X * VETTORE_X) + (VETTORE_Y * VETTORE_Y)) in AFSTAND
            zet 0 in UITREKKING
            als (AFSTAND > 0.1) dan ( zet AFSTAND - L_RUST in UITREKKING )
            zet K_VEER * UITREKKING in VEERKRACHT_MAG

            zet 0 in DIR_X
            zet 0 in DIR_Y
            als (AFSTAND > 0) dan (
                zet VETTORE_X / AFSTAND in DIR_X
                zet VETTORE_Y / AFSTAND in DIR_Y
            )

            zet (VEERKRACHT_MAG * DIR_X) - (C_DEMPER * T_VX) in TOT_KRACHT_X
            zet (VEERKRACHT_MAG * DIR_Y) - (C_DEMPER * T_VY) in TOT_KRACHT_Y
            zet TOT_KRACHT_X / DRIEHOEK_MASSA in ACCEL_X
            zet TOT_KRACHT_Y / DRIEHOEK_MASSA in ACCEL_Y
            zet T_VX + (ACCEL_X * DT) in T_VX_NEW
            zet T_VY + (ACCEL_Y * DT) in T_VY_NEW
            zet T_X + (T_VX_NEW * DT) in T_X_NEW
            zet T_Y + (T_VY_NEW * DT) in T_Y_NEW

            zet T_X_NEW in element i van LIJST_DRIEHOEK_X
            zet T_Y_NEW in element i van LIJST_DRIEHOEK_Y
            zet T_VX_NEW in element i van LIJST_DRIEHOEK_VX
            zet T_VY_NEW in element i van LIJST_DRIEHOEK_VY

            # --- Botsing Speler-Driehoek ---

            zet T_Y_NEW - Y_POS in AFST_TY
            zet (AFST_TX * AFST_TX) + (AFST_TY * AFST_TY) in AFST_T_K
            zet SPELER_STRAAL + DRIEHOEK_STRAAL in R_TOT_T
            zet R_TOT_T * R_TOT_T in R_TOT_T_K

            als (AFST_T_K < R_TOT_T_K) dan (
                zet LEVENS - 1 in LEVENS
                als (LEVENS <= 0) dan (
                    zet 1 in GAME_OVER
                ) anders (
                    schrijf "LEVEN VERLOREN! Levens over: "
                    schrijf LEVENS
                    nieuwe regel
                    zet 0 in X_POS
                    zet 0 in Y_POS
                    teller j loopt van 1 tot NUM_DRIEHOEKEN en je doet (
                        zet willekeurig(-SPEELGEBIED, SPEELGEBIED) in element j van LIJST_DRIEHOEK_X
                        zet willekeurig(-SPEELGEBIED, SPEELGEBIED) in element j van LIJST_DRIEHOEK_Y

                        zet 0 in element j van LIJST_DRIEHOEK_VY

            )
        )
    )

    # --- 3. Botsingslogica Voedsel ---

11. Koch-sneeuwvlok

De Koch-sneeuwvlok is een van de meest bekende fractale krommen.
Dit programma tekent het met behulp van recursie: elke rechte lijn is verdeeld in vier segmenten.
De diepte van de recursie bepaalt de mate van detail van de uiteindelijke fractaal.

# Programma om de Koch-sneeuwvlok-fractaal te tekenen
#
# We definiëren het detailniveau van de fractaal.
# Een waarde van 4 of 5 is een goed begin.
# Hogere waarden (6+) kunnen erg traag zijn.
zet 6 in maxNiveau

# We stellen de lijndikte en de kleur in
zet 1 in lijnDikte
stel kleur in (0, 40, 80) # Een donkerblauw

# We berekenen de constanten voor de rotatie van 60 graden (PI/3 radialen)
# De interpreter gebruikt Math.sin/cos, die radialen vereisen.
zet PI / 3 in PIDERDE
zet cos(PIDERDE) in cos60
zet -sin(PIDERDE) in sin60

#
# Dit is de recursieve procedure die een Koch-kromme tekent
# tussen twee punten (x1, y1) en (x2, y2).
#
definieer procedure tekenKoch(niveau, x1, y1, x2, y2) (
    
    # Basisscenario: als het niveau 0 is, teken dan een rechte lijn.
    als (niveau == 0) dan (
        lijn dikte lijnDikte van (x1, y1) tot (x2, y2)
    )
    
    # Recursief scenario: bereken de 4 segmenten en herhaal.
    anders (
        # 1. Bereken de vector van punt 1 naar punt 2
        zet x2 - x1 in dx
        zet y2 - y1 in dy
        
        # 2. Vind de punten A en B, die de lijn in 3 delen verdelen
        zet x1 + dx / 3 in ax
        zet y1 + dy / 3 in ay
        
        zet x1 + dx * 2 / 3 in bx
        zet y1 + dy * 2 / 3 in by
        
        # 3. Bereken het punt C (de "punt") door het segment A->B
        #    60 graden rond het punt A te draaien.
        zet (dx / 3) * cos60 - (dy / 3) * sin60 in cx_rel
        zet (dx / 3) * sin60 + (dy / 3) * cos60 in cy_rel
        
        zet ax + cx_rel in cx
        zet ay + cy_rel in cy
        
        # 4. Roep de procedure 4 keer op voor de nieuwe segmenten
        tekenKoch(niveau - 1, x1, y1, ax, ay)  # P1 -> A
        tekenKoch(niveau - 1, ax, ay, cx, cy)  # A -> C
        tekenKoch(niveau - 1, cx, cy, bx, by)  # C -> B
        tekenKoch(niveau - 1, bx, by, x2, y2)  # B -> P2
    )
)
